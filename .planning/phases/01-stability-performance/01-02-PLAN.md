```typescript
/**
 * Truncate text to approximate token budget.
 * Uses 4 chars/token heuristic for English text.
 */
export function truncateToTokenBudget(
  text: string,
  maxTokens: number = LIMITS_AI.maxVoiceTokens
): string {
  const maxChars = maxTokens * 4
  if (text.length <= maxChars) return text
  return text.slice(0, maxChars).trimEnd() + '...'
}
```
Note: LIMITS_AI.maxVoiceTokens (200) was added in Plan 01. If it was named differently, match the actual field name.
